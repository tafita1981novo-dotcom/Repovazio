async function upload(idx, url, fname, mime) {
  const r = await fetch(url), b = await r.blob();
  const f = new File([b], fname, {type: mime || b.type || 'image/jpeg'});
  const dt = new DataTransfer(); dt.items.add(f);
  const inp = document.querySelectorAll('input[type="file"]');
  if (!inp[idx]) return 'ERR:no_input_' + idx;
  Object.defineProperty(inp[idx], 'files', {value: dt.files, writable: false});
  inp[idx].dispatchEvent(new Event('change', {bubbles: true}));
  await new Promise(r => setTimeout(r, 2000));
  return 'OK:' + fname;
}
